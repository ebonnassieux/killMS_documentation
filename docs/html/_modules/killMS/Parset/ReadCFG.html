def FormatDico(DicoIn) -> dict:
    '''
    Wrapper for FormatValue function, takes in a dicomodel and returns an ordered dicomodel.
    returns OrderedDict dico with either DicoIn values, or None values.
    
    :param DicoIn: DicoModel object

    '''
    Dico=OrderedDict()
    for key in DicoIn.keys():
        Dico[key]=FormatValue(DicoIn[key])
    return Dico

def FormatValue(ValueIn,StrMode=False) :
    '''
    Function to reformat input values.

    :param ValueIn: Value to be reformated
    :param StrMode: boolean. If True, do not reformat the input.
    '''

    if ValueIn==None: return None


    if "#" in ValueIn:
        # Remove comments
        ValueIn=ValueIn.split("#")[0]

    
    if ('"' in ValueIn)|("'" in ValueIn):
        # Remove whitespace and ', " characters
        Value=ValueIn.replace(" ","").replace('"',"")
        Value=ValueIn.replace(" ","").replace("'","")

    if StrMode:
        # do nothing if StrMode set to True
        return ValueIn

    # check if value is integer
    MayBeInt=False
    if not("." in ValueIn): MayBeInt=True
    if "True" in ValueIn:
        # check if value is boolean true
        Value=True
    elif "False" in ValueIn:
        # check if value is boolean false
        Value=False
    elif "None" in ValueIn:
        # check if value is None
        Value=None
    elif ("[" in ValueIn):
        # remove array [] prefix+suffix in Value string
        Value0=ValueIn[1:-1].split(",")
        try:
            # try recovering array of floats
            Value=[float(v) for v in Value0]
        except:
            # if it fails, just get first value in array
            Value=Value0
    elif ("," in ValueIn):
        # do the same with csv input
        Value0=ValueIn.split(",")
        try:
            # try recovering array of floats
            Value=[float(v) for v in Value0]
        except:
            # just keep first value
            Value=Value0
        
    else:
        try:
            Value=float(ValueIn)
            if MayBeInt: Value=int(Value)
        except:
            Value=ValueIn
            Value=Value.replace(" ","")
    return Value

    def update (self, other) -> None:
        '''
        Updates this Parset object with all keys and values found in provided parset object
        
        :param self: Parset object
        :param other: Parset object
        '''
        for secname, secmap in other.DicoPars.iteritems():
            if secname in self.DicoPars:
                self.DicoPars[secname].update(secmap)
            else:
                self.DicoPars[secname] = secmap

    def Read(self) -> None:
        '''
        Reads a config file. Depends on configparser and ConfigSectionMap.
        Creates the following self objects:
        self.Config
        self.DicoPars
        
        :param self: Parset object
        '''
        config = configparser.ConfigParser(dict_type=OrderedDict)
        config.optionxform = str
        L=config.read(self.File)
        self.Success=True
        if len(L)==0:
            self.Success=False
            return
        self.Config=config
        DicoPars=OrderedDict()
        LSections=config.sections()

        for Section in LSections:
            DicoPars[Section]=self.ConfigSectionMap(Section)
        self.DicoPars=DicoPars

    def ConfigSectionMap(self,section) -> dict:
        '''
        used by Read to do section-by-section initialisation of the param OrderedDicts
        
        :param self: Description
        :param section: Description
        '''
        dict1 = OrderedDict()
        Config=self.Config
        options = Config.options(section)
        for option in options:

            Val=Config.get(section, option)
            Val=Val.replace(" ","")
            Val=Val.replace("'","")
            Val=Val.replace('"',"")
            FVal=FormatValue(Val)#,StrMode=True)

            dict1[option] = FVal
            # if dict1[option] == -1:
            #     DebugPrint("skip: %s" % option)
        return dict1
