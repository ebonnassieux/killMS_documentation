def PrintOptParse(Obj,
                  ValObj,
                  RejectGroup=[]) -> None:
    '''
    Function to print OptParse dictionary values.
    
    :param Obj: Object, first output of MyPickle invocation
    :param ValObj: Object values, second output of MyPickle invocation
    :param RejectGroup: Array including titles to skip the print of.
    '''
    P=ClassPrint.ClassPrint(HW=50)
    LGroups=Obj.option_groups
    print(ModColor.Str(" Selected Options:"))

    for Group in LGroups:
        Skip=False
        for Name in RejectGroup:
            if Name in Group.title:
                Skip=True

        if Skip: continue
        print(ModColor.Str(Group.title,col="green"))

        option_list=Group.option_list
        for o in option_list:
            lopt=o._long_opts[0]
            oname=lopt.split("--")[-1]


            V=getattr(ValObj,oname)
            if True:#V!="":

                if V=="": V="''"

                #P.Print(oname,V)
                default=o.default
                H=o.help.replace("%default",str(default))
                #P.Print2(oname,V,H)
                P.Print(oname,V)

                # strName=%s
                # print "       "oname,V
        print()
