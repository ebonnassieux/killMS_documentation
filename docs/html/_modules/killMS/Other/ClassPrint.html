class ClassPrint():
    '''
    Class that defines the terminal print parameters.
    '''
    def __init__(self,
                 HW:int=20,
                 quote:str='"'):
        '''
        Docstring for __init__
        
        :param self: ClassPrint
        :param HW: Header Width (?)
        :type HW: int
        :param quote: String to use for quotes.
        :type quote: str
        '''
        self.LeftW=HW
        self.WV0=50
        self.WV1=30
        self.WidthHelp=30
        self.quote=quote
        self.proto=" - %s %s %s : "
        self.Lproto=len(self.proto.replace("%s",""))

    def Print(self,
              par:str,
              value:str,
              value2=None) -> None:
        '''
        Print function to align logger outputs.
        
        :param self: ClassPrint
        :param par: Parameter to print value of
        :type par: str
        :param value: Value of the parameter
        :type value: str
        :param value2: Optional second parameter value
        '''
        # align the param
        parout=" - %s %s"%(par,"."*(self.LeftW-len(par)))
        if value2==None:
            valueOut=value
        else:
            valueOut="%s%s"%(value.ljust(self.WV0),(""" "%s" """%value2).rjust(self.WV1))
        print("%s = %s"%(parout,valueOut))

    def Print2(self,
               par:str,
               value:str,
               helpit:str="",
               col="white") -> None:
        '''
        Print function to align logger outputs, with colourful formatting.
        Prints out pretty logger.

        :param self: ClassPrint
        :param par: parameter to print value of
        :type par: str
        :param value: Value of the parameter
        :type value: str
        :param helpit: Help string to be printed. Default is no help given.
        :type helpit: str
        :param col: Colour of the print. Default is white.
        :type col: str
        '''
        WidthTerm=self.getWidth()
        Lpar=len(str(par))
        Lval=len(str(value))
        SFill="."*max(self.LeftW-self.Lproto-Lpar-Lval,10)
        WidthHelp=WidthTerm-(self.Lproto+Lpar+Lval+len(SFill))
        Spar="%s"%ModColor.Str(str(par),col=col,Bold=False)
        Sval="%s"%ModColor.Str(str(value),col=col,Bold=False)
        if helpit=="":
            helpit="Help yourself"
        Shelp="%s"%helpit
        if WidthHelp<0:
             print(self.proto%(Spar,SFill,Sval)+Shelp)
             return
        Lhelp=len(str(helpit))
        listStrHelp=range(0,Lhelp,WidthHelp)
        
        if listStrHelp[-1]!=Lhelp: listStrHelp.append(Lhelp)
        
        
        
        print(self.proto%(Spar,SFill,Sval)+Shelp[0:WidthHelp])
        for i in range(1,len(listStrHelp)-1):
            parout="%s: %s"%(" "*(self.LeftW-2),Shelp[listStrHelp[i]:listStrHelp[i+1]])
            print(parout)
